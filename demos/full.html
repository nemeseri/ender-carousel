    Pager

    Example HTML

    <div class="carousel">
    <div class="window">
        [...]
    </div> <!-- .window -->

    <div id="pager">
        <ul></ul>
    </div>

    <a class="prev">Previous</a>
    <a class="next">Next</a>
</div> <!-- .carousel -->

    Example CSS

    #pager {
    display: block;
    position: relative;
    float: left;
    left: 50%;
    margin-top: 10px;
}

#pager ul {
    position: relative;
    float: left;
    left: -50%;
    font-size: 0;
    overflow: hidden;
}

#pager li {
    display: inline-block;
    width: 10px;
    height: 10px;
    list-style: none;
    background-color: #aaa;
}

#pager .active {
    background-color: #cc6600;
}

    Active item(s)

    Example CSS

    .window li {
    opacity: 0.4;
}

.window .active {
    opacity: 1;
}

    Initialize the ender-carousel plugin

    To enable the pager use the 'pager' property, to enable active item(s) class names use 'activeClass'.

    $(document).ready(function () {
    $(".carousel").carousel({
        pager: "#pager ul",
        activeClass: "active",
        keyboard: true
    });
});
